🤖 AI-powered cultural recommendations (Google Gemini)

🔌 Frontend-Backend Integration
1. API Client Setup (frontend/services/api/client.ts)
TypeScript
Copy
// Axios instance with interceptors
import axios from 'axios';
import { useAuthStore } from '@/stores/auth-store';

const apiClient = axios.create({
  baseURL: process.env.EXPO_PUBLIC_API_URL || 'http://localhost:8000',
  timeout: 10000,
  headers: {
    'Content-Type': 'application/json',
  },
});

// Request interceptor - add auth token
apiClient.interceptors.request.use((config) => {
  const token = useAuthStore.getState().accessToken;
  if (token) {
    config.headers.Authorization = `Bearer ${token}`;
  }
  return config;
});

// Response interceptor - handle token refresh
apiClient.interceptors.response.use(
  (response) => response,
  async (error) => {
    const originalRequest = error.config;
    
    if (error.response?.status === 401 && !originalRequest._retry) {
      originalRequest._retry = true;
      const refreshToken = useAuthStore.getState().refreshToken;
      
      // Call refresh endpoint
      const response = await axios.post('/auth/refresh', {
        refresh_token: refreshToken,
      });
      
      // Update tokens
      useAuthStore.getState().setTokens(response.data);
      
      // Retry original request
      originalRequest.headers.Authorization = `Bearer ${response.data.access_token}`;
      return apiClient(originalRequest);
    }
    
    return Promise.reject(error);
  }
);
2. WebSocket Integration (frontend/services/websocket.ts)
TypeScript
Copy
// React Native WebSocket service
class WebSocketService {
  private ws: WebSocket | null = null;
  private reconnectAttempts = 0;
  private maxReconnectAttempts = 5;

  connect(token: string) {
    const wsUrl = `${process.env.EXPO_PUBLIC_WS_URL}/ws/messaging?token=${token}`;
    this.ws = new WebSocket(wsUrl);

    this.ws.onopen = () => {
      console.log('WebSocket connected');
      this.reconnectAttempts = 0;
    };

    this.ws.onmessage = (event) => {
      const data = JSON.parse(event.data);
      this.handleMessage(data);
    };

    this.ws.onclose = () => {
      if (this.reconnectAttempts < this.maxReconnectAttempts) {
        setTimeout(() => this.connect(token), 3000);
        this.reconnectAttempts++;
      }
    };
  }

  joinConversation(conversationId: string) {
    this.send({
      action: 'join_conversation',
      conversation_id: conversationId,
    });
  }

  sendMessage(conversationId: string, content: string, type = 'text') {
    this.send({
      action: 'send_message',
      conversation_id: conversationId,
      content,
      message_type: type,
    });
  }

  private send(data: object) {
    if (this.ws?.readyState === WebSocket.OPEN) {
      this.ws.send(JSON.stringify(data));
    }
  }
}

export const wsService = new WebSocketService();
3. Hook Integration Example (frontend/hooks/use-ai-recommendations.ts)
TypeScript
Copy
import { useQuery } from '@tanstack/react-query';
import { apiClient } from '@/services/api/client';

interface CulturalRecommendationParams {
  location: string;
  interests: string[];
  travelStyle: 'relaxed' | 'balanced' | 'packed';
  budget?: string;
  durationDays: number;
}

export function useAIRecommendations(params: CulturalRecommendationParams) {
  return useQuery({
    queryKey: ['cultural-recommendations', params],
    queryFn: async () => {
      const response = await apiClient.post('/cultural/recommendations', {
        location: params.location,
        interests: params.interests,
        travel_style: params.travelStyle,
        budget: params.budget,
        duration_days: params.durationDays,
      });
      return response.data;
    },
    enabled: !!params.location,
    staleTime: 1000 * 60 * 5, // 5 minutes
  });
}
4. Authentication Store (frontend/stores/auth-store.ts)
TypeScript
Copy
import { create } from 'zustand';
import { persist, createJSONStorage } from 'zustand/middleware';
import AsyncStorage from '@react-native-async-storage/async-storage';

interface AuthState {
  user: User | null;
  accessToken: string | null;
  refreshToken: string | null;
  isAuthenticated: boolean;
  
  // Actions
  setTokens: (tokens: TokenResponse) => void;
  login: (email: string, password: string) => Promise<void>;
  logout: () => void;
  register: (data: RegisterRequest) => Promise<void>;
}

export const useAuthStore = create<AuthState>()(
  persist(
    (set, get) => ({
      user: null,
      accessToken: null,
      refreshToken: null,
      isAuthenticated: false,

      setTokens: (tokens) => {
        set({
          accessToken: tokens.access_token,
          refreshToken: tokens.refresh_token,
          isAuthenticated: true,
        });
      },

      login: async (email, password) => {
        const response = await apiClient.post('/auth/login', {
          email,
          password,
        });
        get().setTokens(response.data);
        // Fetch user profile
        const userResponse = await apiClient.get('/auth/me');
        set({ user: userResponse.data });
      },

      logout: () => {
        set({
          user: null,
          accessToken: null,
          refreshToken: null,
          isAuthenticated: false,
        });
      },
    }),
    {
      name: 'auth-storage',
      storage: createJSONStorage(() => AsyncStorage),
    }
  )
);
